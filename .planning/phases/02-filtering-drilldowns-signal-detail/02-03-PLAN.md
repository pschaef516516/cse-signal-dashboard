1. Create `src/components/charts/EnrollmentUpsellSplitChart.jsx` with EXACTLY this content (mirrors SignalVolumeChart but renders only enrollment + upsell):

```jsx
import {
  BarChart, Bar, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer,
} from 'recharts'
import { groupByWeek } from '../../utils/aggregate'

// Per D-13: stacked bar showing enrollment vs upsell volume over time.
// Data shape from groupByWeek: { week, churn, enrollment, upsell } — we render only enrollment + upsell.
export default function EnrollmentUpsellSplitChart({ signals }) {
  const data = groupByWeek(signals)

  if (data.length === 0) {
    return <p className="text-sm text-gray-400">No data yet.</p>
  }

  return (
    <ResponsiveContainer width="100%" height={300}>
      <BarChart data={data} margin={{ top: 4, right: 16, left: 0, bottom: 4 }}>
        <CartesianGrid strokeDasharray="3 3" stroke="#f0f0f0" />
        <XAxis dataKey="week" tick={{ fontSize: 11 }} />
        <YAxis allowDecimals={false} tick={{ fontSize: 11 }} />
        <Tooltip />
        <Legend />
        <Bar dataKey="enrollment" stackId="a" fill="#0057FF" name="Enrollment" />
        <Bar dataKey="upsell" stackId="a" fill="#623CC9" name="Upsell" />
      </BarChart>
    </ResponsiveContainer>
  )
}
```

2. Create `src/components/charts/EUCommunityChart.jsx` with EXACTLY this content (mirrors CommunityChart's horizontal layout but uses `groupBySourceAndType` and renders TWO bars per source):

```jsx
import {
  BarChart, Bar, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer,
} from 'recharts'
import { groupBySourceAndType } from '../../utils/aggregate'

// Per D-12: source breakdown for E&U signals — split by enrollment vs upsell per source.
// Data shape from groupBySourceAndType: [{ name, enrollment, upsell }, ...] sorted by total desc.
export default function EUCommunityChart({ signals }) {
  const data = groupBySourceAndType(signals)

  if (data.length === 0) {
    return <p className="text-sm text-gray-400">No data yet.</p>
  }

  return (
    <ResponsiveContainer width="100%" height={300}>
      <BarChart
        layout="vertical"
        data={data}
        margin={{ top: 4, right: 32, left: 8, bottom: 4 }}
      >
        <CartesianGrid strokeDasharray="3 3" stroke="#f0f0f0" horizontal={false} />
        <XAxis type="number" allowDecimals={false} tick={{ fontSize: 11 }} />
        <YAxis type="category" dataKey="name" width={160} tick={{ fontSize: 11 }} />
        <Tooltip />
        <Legend />
        <Bar dataKey="enrollment" fill="#0057FF" name="Enrollment" />
        <Bar dataKey="upsell" fill="#623CC9" name="Upsell" />
      </BarChart>
    </ResponsiveContainer>
  )
}
```
